ğŸ§° Starter Template

Use this as a base:

<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>My Personal Page</title>
</head>
<body>
  <header>
    <h1>Welcome!</h1>
  </header>

  <nav>
    <a href="#about">About</a>
    <a href="#contact">Contact</a>
  </nav>

  <main>
    <section id="about">
      <h2>About Me</h2>
      <p>I love building websites and learning HTML!</p>
      <img src="me.jpg" alt="My picture" />
    </section>

    <section>
      <h2>My Favorites</h2>
      <ul>
        <li>Coding</li>
        <li>Design</li>
        <li>Coffee</li>
      </ul>
    </section>

    <section id="contact">
      <h2>Contact Me</h2>
      <form>
        <label for="email">Email:</label>
        <input type="email" id="email" name="email" required />
        <label for="message">Message:</label>
        <textarea id="message" name="message"></textarea>
        <button type="submit">Send</button>
      </form>
    </section>
  </main>

  <footer>
    <p>© 2025 YourName. All rights reserved.</p>
  </footer>
</body>
</html>
